int buzzerpin = 11;
int bpm = 120;
void setup() {
pinMode(buzzerpin, OUTPUT);
}
void t(int frequence, float duration)
{
tone(buzzerpin, frequence);
delay(1000 * (bpm / 60) * duration);
noTone(buzzerpin);
delay(duration * 1.3);
}
void p(float duration)
{
delay(1000 * ((bpm / 60) / duration));
}
